import { ConfigService } from './configService.js'
import { API_ENDPOINTS } from '../utils/constants.js'

export class AIService {
  constructor() {
    this.config = ConfigService.getAIConfig()
  }
  
  async sendMessage(message, options = {}) {
    const { stream = true, context = [] } = options
    
    const requestBody = {
      model: this.config.model,
      messages: [
        ...context,
        { role: 'user', content: message }
      ],
      stream
    }
    
    if (stream) {
      return this.sendStreamMessage(requestBody)
    } else {
      return this.sendSyncMessage(requestBody)
    }
  }
  
  async sendStreamMessage(requestBody) {
    const response = await fetch(`${this.config.baseUrl}${API_ENDPOINTS.CHAT}`, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'Authorization': `Bearer ${this.config.apiKey}`
      },
      body: JSON.stringify(requestBody)
    })
    
    if (!response.ok) {
      throw new Error(`HTTP error! status: ${response.status}`)
    }
    
    return this.processStreamResponse(response)
  }
  
  async processStreamResponse(response) {
    const reader = response.body.getReader()
    const decoder = new TextDecoder()
    let result = ''
    
    try {
      while (true) {
        const { done, value } = await reader.read()
        if (done) break
        
        const chunk = decoder.decode(value)
        const lines = chunk.split('\n')
        
        for (const line of lines) {
          if (line.startsWith('data: ')) {
            const data = line.slice(6)
            if (data === '[DONE]') return result
            
            try {
              const parsed = JSON.parse(data)
              const content = parsed.choices?.[0]?.delta?.content
              if (content) {
                result += content
                // è§¦å‘æµå¼æ›´æ–°äº‹ä»¶
                EventBus.emit('ai:stream', { content, fullContent: result })
              }
            } catch (e) {
              console.warn('è§£ææµå¼æ•°æ®å¤±è´¥:', e)
            }
          }
        }
      }
    } finally {
      reader.releaseLock()
    }
    
    return result
  }
}

export class Router {
  constructor() {
    this.routes = new Map()
    this.currentRoute = null
    
    this.init()
  }
  
  init() {
    window.addEventListener('hashchange', this.handleRouteChange.bind(this))
    window.addEventListener('load', this.handleRouteChange.bind(this))
  }
  
  register(path, component) {
    this.routes.set(path, component)
  }
  
  navigate(path) {
    window.location.hash = path
  }
  
  handleRouteChange() {
    const hash = window.location.hash.slice(1) || '/'
    const component = this.routes.get(hash)
    
    if (component) {
      this.currentRoute = hash
      this.renderComponent(component)
    } else {
      this.navigate('/')
    }
  }
  
  renderComponent(component) {
    const app = document.getElementById('app')
    if (typeof component === 'function') {
      component(app)
    } else if (component.render) {
      component.render(app)
    }
  }
}
